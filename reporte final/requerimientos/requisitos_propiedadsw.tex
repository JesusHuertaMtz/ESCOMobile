\subsection{Requsitos de la propiedad de software}

\begin{requisitos}{RPS Presentación del mapa de ESCOM.} 
	\RFitem{Descripción}{El mapa de ESCOM que mostrará la aplicación es en 2D, la vista será aérea 
	y mostrará texto que describe cada área.}
	\RFitem{Propósito}{Hacer la navegación en el mapa sencilla para el actor.} 
	\RFitem{Usuario}{Alumno, Profesor.}
	\RFitem{Tipo}{Móvil.} 
\end{requisitos}

\begin{requisitos}{RPS Mapa de ESCOM en móviles.} 
	\RFitem{Descripción}{El mapa de ESCOM solo estará disponible en la versión móvil de la aplicación.} 
	\RFitem{Propósito}{Atraer a los estudiantes y profesores de ESCOM a usar ESCOMobile.} 
	\RFitem{Usuario}{N/A.} 
	\RFitem{Tipo}{Móvil.} 
\end{requisitos}

\begin{requisitos}{RPS Personalización Perfiles.} 
	\RFitem{Descripción}{La aplicación permite configurar y personalizar los perfiles con ajustes menores
	(foto de perfil, algún comentario, etc) de los alumnos. } 
	\RFitem{Propósito}{}.
	\RFitem{Usuario}{Alumno.}
	\RFitem{Tipo}{Móvil.} 
\end{requisitos}

\begin{requisitos}{RPS Aplicación sin conexión a Internet.} 
	\RFitem{Descripción}{El sistema ofrece un modo 'sin conexión', mismo que permite al usuario
	visualizar ciertos servicios de la app (por medio del caché) sin necesidad de tener una conexión a
	Internet.} 
	\RFitem{Propósito}{El usuario continúa usando servicios de la app incluso sin conexión.} 
	\RFitem{Usuario}{Alumno, Profesor, Visitante.}
	\RFitem{Tipo}{Móvil.} 
\end{requisitos}

\begin{requisitos}{Cifrado de información.} 
	\RFitem{Descripción}{La información importante del usuario es cifrada mediante AES 256.} 
	\RFitem{Propósito}{Brindar al usuario seguridad y confianza al proteger sus datos. } 
	\RFitem{Usuario}{Alumno, profesor.} 
	\RFitem{Tipo}{Web / móvil.} 
\end{requisitos}
