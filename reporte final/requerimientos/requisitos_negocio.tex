\subsection{Requisitos de negocio}


\begin{requisitos}{RN Mapa de ESCOM.} 
	\RFitem{Descripción}{La aplicación tendrá un módulo para consultar el mapa de la ESCOM. Se 
	mostrará una vista aérea de los pisos de los edificios.}
	\RFitem{Propósito}{Dar una vista general de los edificios que conforman al plantel y orientar a
	alumnos a ubicar donde se encuentran ubicados los salones, cubículos, clubes, biblioteca 
	y áreas administrativas en ESCOM.}
	\RFitem{Usuario}{Alumno, Profesor.} 
	\RFitem{Tipo}{Móvil.} 
\end{requisitos}


\begin{requisitos}{RN Actualización del programa de eventos.} 
	\RFitem{Descripción}{El sistema permitirá añadir, actualizar y eliminar eventos que ocurran
	dentro de las instalaciones de la ESCOM. } 
	\RFitem{Propósito}{El usuario conoce los eventos a realizarse en los días consecutivos. }
	\RFitem{Usuario}{Jefe del departamento de extensión y apoyos educativos.} 
	\RFitem{Tipo}{Web.} 
\end{requisitos}

\begin{requisitos}{RN Campos obligatorios.} 
	\RFitem{Descripción}{El sistema válida que no se omitan los campos marcados como obligatorios.}
	\RFitem{Propósito}{No se omita información necesaria e importante para el sistema. } 
	\RFitem{Usuario}{Alumno, profesor.} 
	\RFitem{Tipo}{Móvil.} 
\end{requisitos}

\begin{requisitos}{RN Validación de la información.}
	\RFitem{Descripción}{El sistema válida que los datos introducidos en cada campo cumplan con un
	formato determinado.} 
	\RFitem{Propósito}{Evitar inconsistencia de datos.} 
	\RFitem{Usuario}{Alumno, profesor.} 
	\RFitem{Tipo}{Móvil.} 
\end{requisitos}

\begin{requisitos}{RN Acciones del alumno.} 
	\RFitem{Descripción}{El alumno contará con la posibilidad de consultar y modificar su información, consultas sus citas, eventos, bolsa de trabajo, consultar áreas escolares a través del mapa, consultar profesores y su información, cerrar su sesión, eliminar su cuenta, etc.}
	\RFitem{Propósito}{Que el usuario tengo acceso a la información escolar.} 
	\RFitem{Usuario}{Alumno.}
	\RFitem{Tipo}{Móvil.} 
\end{requisitos}

\begin{requisitos}{RN Acciones del profesor.} 
	\RFitem{Descripción}{El profesor contará con la posibilidad de consultar y modificar su información, consultas sus citas, eventos, consultar áreas escolares a través del mapa, cerrar su sesión, eliminar su cuenta, etc.}
	\RFitem{Propósito}{Que el usuario tengo acceso a la información escolar.} 
	\RFitem{Usuario}{Profesor.}
	\RFitem{Tipo}{Móvil.} 
\end{requisitos}
